// Helper Function to Unlock a Stage function unlockStage(elementId, destinationFile) { const card = document.getElementById(elementId); card.classList.remove('locked-state'); card.classList.add('active-card'); const btn = card.querySelector('button'); btn.removeAttribute('disabled'); btn.className = "w-full bg-green-600 text-white py-4 rounded-2xl font-bold hover:bg-green-700 transition shadow-lg cursor-pointer"; btn.innerText = "Start Stage"; btn.onclick = function() { window.location.href = destinationFile; }; } // Updated Helper: Changes Badge based on score (Decreasing Reward) function unlockBadge(index, score) { const badgeBox = document.getElementById('badges-container').children[index]; const badgeIcon = badgeBox.querySelector('div'); const badgeText = badgeBox.querySelector('p'); badgeBox.classList.remove('opacity-30', 'grayscale'); badgeIcon.classList.remove('border-dashed', 'border-gray-300'); badgeIcon.classList.add('border-solid', 'shadow-sm'); // Logic for decreasing rewards (Visual/Text) if (score >= 4) { // Gold/High Performance badgeIcon.classList.add('border-yellow-400', 'bg-yellow-50'); badgeBox.classList.add('badge-unlock'); // Animation badgeText.classList.replace('text-gray-400', 'text-yellow-700'); } else if (score >= 2) { // Silver/Medium Performance badgeIcon.classList.add('border-green-500', 'bg-green-50'); badgeText.classList.replace('text-gray-400', 'text-green-700'); } else { // Bronze/Low Performance (Decreased Reward) badgeIcon.classList.add('border-orange-300', 'bg-orange-50'); badgeText.classList.replace('text-gray-400', 'text-orange-600'); badgeText.innerText += " (Trainee)"; } }
